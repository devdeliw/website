 80class EDMRLeastSquares(): 
 81    """ 
 82    Performs Least-Squares Optimization to fit the derived EDMR model 
 83    (see ...solving_SLE/main/) against raw EDMR experimental data. 
 84
 85    """
 86    def __init__( 
 87            self, 
 88            raw_B_array: np.ndarray, 
 89            raw_I_array: np.ndarray,
 90            *, 
 91            n_points_interp: int                        = 100, 
 92            alpha: float                                = 1.0,
 93            show_progress: bool                         = True, 
 94            update_param_yaml: bool                     = False,
 95            init_check: bool                            = True, 
 96            modulate: bool                              = True, 
 97            lsq_params: LeastSquaresParameters          = LeastSquaresParameters(), 
 98            multithread_params: LeastSquaresMultiThread = LeastSquaresMultiThread(), 
 99            pickle_params: LeastSquaresPickling         = LeastSquaresPickling(), 
100    ): 
101        """
102        Every parameter below `alpha` is insignificant; 
103        Assuming you update parameters via the params.yaml file. 
104        Using the default values for the rest are fine. 
105
106        Args: 
107            * raw_B_array: np.ndarray
108                * Raw EDMR B-field Sweep Data 
109            * raw_I_array: np.ndarray 
110                * Raw EDMR measured current 
111
112            ~/ Keyword only ~/ 
113
114            * n_points_interp: int 
115                * Number of points to discretize B by for simulating
116                * Reduces complexity by simulating less B values  
117                * with cubic-spline interpolation for residuals. 
118                * A good heuristic is 1-point / Gauss
119            * alpha: float
120                * Residual weight exponent to emphasize derivative 
121            * show_progress: bool 
122                * Display tqdm progress bar.
123            * update_param_yaml: bool 
124                * Update the parameter file with the latest stored pickle 
125                * from a previous run. Uses these params as well.
126
127        Variable Notation: 
128            * `parameters` is a dict[str, float] of parameter names and values 
129            * `parameter_vec` is a np.ndarray built from the `parameters` dict
130        """
131
132        self._lsq = lsq_params 
133        self._mt  = multithread_params 
134        self._pkl = pickle_params
135
136        self._save_pickle = self._pkl.save_pickle  
137        self._pickle_dirs = self._pkl.pickle_dirs
138        self._pickle_dirs.mkdir(exist_ok=True, parents=True)
139
140        # Initial guess parameters 
141        if self._lsq.ansatz_parameters is None or self._lsq.default_parameters: 
142            ansatz_parameters = load_params(PARAMETER_FILE) 
143        else: 
144            ansatz_parameters = self._lsq.ansatz_parameters
145        if update_param_yaml: 
146            _update_yaml_with_pickle(self)
147            ansatz_parameters = load_params(PARAMETER_FILE)
148        self.verify_parameters_complete(ansatz_parameters)
149 
150        # Some "parameters" are constants, h, hbar, mu_B, mu_N.
151        # Since these are part of the parameter vector, the LSQ routine 
152        # would run the EDMR simulation 4 more times than needed *every* iteration. 
153        #
154        # Hence we remove constants from ansatz_parameters and reinsert them 
155        # when calculating the EDMR simulated spectra.  
156       
157        lower_bound = self._lsq.lower_bound 
158        upper_bound = self._lsq.upper_bound 
159        assert lower_bound.keys() == upper_bound.keys(), \
160            "lower_bound.keys() != upper_bound.keys()"
161        assert upper_bound.keys() == ansatz_parameters.keys(), \
162            "upper_bound.keys() != ansatz_parameters.keys()" 
163
164        keys_with_common_values = [
165            k for k in ansatz_parameters.keys() 
166            if (
167                np.isclose(ansatz_parameters[k],lower_bound[k], atol=1e-11) and 
168                np.isclose(lower_bound[k], upper_bound[k], atol=1e-11)
169            )
170        ]
171        parameters   = {
172            k: v for k, v in ansatz_parameters.items() if k not in keys_with_common_values
173        }
174        lower_bound = {
175            k: v for k, v in lower_bound.items() if k not in keys_with_common_values
176        } 
177        upper_bound = {
178            k: v for k, v in upper_bound.items() if k not in keys_with_common_values
179        } 
180
181        self.constants     = {k: ansatz_parameters[k] for k in keys_with_common_values}
182        self.parameter_vec = _convert_dict_to_theta(parameters)
183        self._lower_bounds = _convert_dict_to_theta(lower_bound)
184        self._upper_bounds = _convert_dict_to_theta(upper_bound) 
185
186        # Raw EDMR experimental data
187        self.raw_B_data = np.asarray(raw_B_array) 
188        self.raw_I_data = np.asarray(raw_I_array) 
189        if len(raw_B_array) > n_points_interp: 
190            self.B_discretized = np.linspace(
191                raw_B_array[0], 
192                raw_B_array[-1], 
193                n_points_interp
194            )
195        else: self.B_discretized = self.raw_B_data
196
197        self.alpha            = alpha
198        self._n_points        = n_points_interp
199        self._show_progress   = show_progress
200        self.modulate         = modulate
201
202        # multithreading
203        self._n_jobs          = self._mt.n_jobs 
204        self._blocks_per_core = self._mt.blocks_per_core 
205
206        # preliminary checks 
207        if not init_check: 
208            print(
209                "WARNING: not performing preliminary checks. "
210                "This is highly recommended and only takes ~0.1s per iteration."
211            )
212        self.init_check = init_check 
213
214        self.edmr_functional = compute_edmr_spectra

227    def _parameter_vec_to_full_dict(self, parameter_vec: np.ndarray): 
228        """ 
229        Converts the ndarray parameter vector returned by least_squares 
230        back into a parameters dict[str, float] with constants included. 
231
232        """
233        parameters = _convert_theta_to_dict(parameter_vec)
234        return self._reinsert_constants_to_parameters(parameters)

236    def compute_residuals(self, parameter_vec: np.ndarray):
237        """ 
238        Args: 
239            * parameters: dict[str, float], 
240                * Model parameters that are undergoing optimization 
241            * alpha: int 
242                * Exponent of derivative residual weighting 
243                * Weighting by the derivative emphasizes EDMR features 
244                * alpha = 0 weights every point equally
245            * save_pickle: bool 
246                * Saves lsq parameters everytime they're updated. 
247        """ 
248        parameters = self._parameter_vec_to_full_dict(parameter_vec)
249        simulated_current = self.edmr_functional( 
250            B_array=self.B_discretized, 
251            parameters=parameters, 
252            modulate=self.modulate, 
253            n_jobs=self._n_jobs, 
254            blocks_per_core=self._blocks_per_core, 
255            show_progress=self._show_progress, 
256            init_check=self.init_check
257        )
258
259        # Interpolating discrete simulated current 
260        # to same length as self.raw_B_data for residuals 
261        I_interpolated = interp1d(
262            self.B_discretized, 
263            simulated_current, 
264            kind="cubic", 
265            assume_sorted=True
266        )(self.raw_B_data) 
267    
268        # Weight residuals by derivative (emphasize features) 
269        w = np.abs(np.gradient(self.raw_I_data, self.raw_B_data)) 
270        w = w ** self.alpha
271        return w * (I_interpolated - self.raw_I_data)

273    def _callback_save_pickle(self, parameter_vec: np.ndarray):
274        """
275        Saves the most recent parameter + constant dictionary in a pickle 
276        after every least_squares iteration. 
277
278        * Only if self._save_pickle == True *
279
280        """
281        if self._save_pickle:
282            parameters = self._parameter_vec_to_full_dict(parameter_vec) 
283            pickle_fname = self._pickle_dirs / "lsq_param.pickle"
284            with open(pickle_fname, "wb") as f: 
285                pickle.dump(parameters, f)
286                logger.info(f"~/ Parameter pickle saved. ~/")

288    def fit(
289        self, 
290        ftol: float = 1e-12, 
291        xtol: float = 1e-12, 
292        gtol: float = 1e-12, 
293    ): 
294        if self._pkl.save_pickle: 
295            logger.info(" ~/ Pickle Saving Active /~") 
296            logger.info(" This will consume ~130us every full iteration.")
297            logger.info(f" {self._pickle_dirs / 'lsq_param.pickle'}")
298
299        start = datetime.now() 
300        print(f"\nLeast-Squares started at {start:%Y‑%m‑%d %H:%M:%S}")
301        print("--------------------------------------------") 
302        print("Ansatz Parameters: ") 
303        print(_convert_theta_to_dict(self.parameter_vec))
304        print(f"xtol = {xtol:2g}") 
305        print(f"ftol = {ftol:2g}")
306        print(f"gtol = {gtol:2g}")
307        print(f"# points  = {self._n_points}" )
308        print(f"# threads = {self._n_jobs}\n") 
309
310        self.result = least_squares( 
311            fun     = self.compute_residuals, 
312            x0      = self.parameter_vec, 
313            bounds  = (self._lower_bounds, self._upper_bounds), 
314            ftol    = ftol, 
315            xtol    = xtol, 
316            gtol    = gtol, 
317            verbose = 2, 
318            jac     = '2-point', 
319            method  = 'trf', 
320            callback= self._callback_save_pickle
321        )
322
323        end = datetime.now() 
324        delta = (end - start).total_seconds()
325        print(f"\nLeast-Squares ended at {end:%Y-%m-%d %H:%M:%S}")
326        print(f"Time Elapsed: {delta}s\n")
327
328        return _convert_theta_to_dict(self.result.x)

520def _build_least_squares_fitter(
521    raw_B_data: np.ndarray,
522    raw_I_data: np.ndarray,
523    *,
524    n_points_interp: int            = 100,
525    B_range: tuple[float, float]    = (-np.inf, np.inf),  
526    show_progress: bool             = False,
527    init_check: bool                = True,
528    lsq_params: LeastSquaresParameters          = LeastSquaresParameters(), 
529    multithread_params: LeastSquaresMultiThread = LeastSquaresMultiThread(), 
530    pickle_params: LeastSquaresPickling         = LeastSquaresPickling(), 
531
532
533) -> EDMRLeastSquares: 
534    """ 
535    Builds a `EDMR_LeastSquares` instance for fitting with default parameters 
536    (from params.yaml). 
537
538    Args: 
539        * raw_B_data, raw_I_data: np.ndarray 
540            * raw EDMR experimental data. 
541        * n_points_interp: int 
542            * How many points to use for simulation (resolution) 
543        * n_jobs: int 
544            * Number of cores for simulation 
545        * blocks_per_core: int
546            * Number of B-blocks per core 
547        * B_range: tuple[float, float] 
548            * B_range to be swept over -- included 
549            * if B_range exceeds range of `raw_B_data`: 
550            *   `raw_B_data` is used. 
551            * if B_range is less than range of `raw_B_data`: 
552            *   `raw_B_data` is truncated to match `B_range`
553        * show_progress: bool 
554            * Display tqdm progress bar 
555    """
556
557    raw_B_data = np.asarray(raw_B_data) 
558    raw_I_data = np.asarray(raw_I_data) 
559    assert len(raw_B_data) == len(raw_I_data), "B and I arrays length mismatch."
560
561    if B_range: 
562        mask = (raw_B_data >= B_range[0]) & (raw_B_data <= B_range[1]) 
563        raw_B_data = raw_B_data[mask] 
564        raw_I_data = raw_I_data[mask] 
565
566    return EDMRLeastSquares( 
567            raw_B_array=raw_B_data,
568            raw_I_array=raw_I_data, 
569            n_points_interp=n_points_interp, 
570            show_progress=show_progress, 
571            init_check=init_check,
572            lsq_params=lsq_params,     
573            multithread_params=multithread_params,
574            pickle_params=pickle_params, 
575            update_param_yaml=False,
